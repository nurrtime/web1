function openPage(pageName, elmn, color) { let i, tabcontent, tablinks; tabcontent = document.getElementsByClassName("tabcontent"); for(i = 0; i < tabcontent.length; i++) { tabcontent[i].style.display = "none"; } tablinks = document.getElementsByClassName("tablink"); for(i = 0; i < tablinks.length; i++) { tablinks[i].style.border = ""; } document.getElementById(pageName).style.display = "flex"; elmn.style.border = "2px solid " + color; } document.getElementById("defaultOpen").click();
